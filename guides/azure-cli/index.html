Azure CLI Installation Guide

Ubuntu / Debian

curl -sL https://aka.ms/InstallAzureCLIDeb | sudo bash
az version

RHEL / CentOS / Fedora

sudo rpm --import https://packages.microsoft.com/keys/microsoft.asc
sudo dnf install -y https://packages.microsoft.com/config/rhel/9.0/packages-microsoft-prod.rpm || true
sudo dnf install -y azure-cli || sudo yum install -y azure-cli
az version

Arch Linux

sudo pacman -Syu --noconfirm azure-cli
az version

openSUSE

sudo rpm --import https://packages.microsoft.com/keys/microsoft.asc
sudo zypper addrepo https://packages.microsoft.com/yumrepos/azure-cli azure-cli
sudo zypper refresh
sudo zypper install -y azure-cli
az version

macOS

brew update && brew install azure-cli
az version

Windows

# winget
winget install --id Microsoft.AzureCLI -e

# choco
choco install azure-cli -y

az version

Login Check

az login
az account show
