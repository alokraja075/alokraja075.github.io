Prometheus Installation Guide

Linux (Ubuntu/Debian/RHEL/Fedora/Arch/openSUSE) binary method

cd /tmp
curl -LO https://github.com/prometheus/prometheus/releases/latest/download/prometheus-*.linux-amd64.tar.gz
# If wildcard download fails, open release page and copy exact filename.

tar -xvf prometheus-*.linux-amd64.tar.gz
cd prometheus-*.linux-amd64
./prometheus --version
./prometheus --config.file=prometheus.yml
# UI: http://localhost:9090

Ubuntu / Debian (package alternative)

sudo apt update
sudo apt install -y prometheus
prometheus --version
sudo systemctl enable --now prometheus

RHEL / Fedora

sudo dnf install -y prometheus || sudo yum install -y prometheus
prometheus --version
sudo systemctl enable --now prometheus

macOS

brew install prometheus
prometheus --version
prometheus --config.file=/opt/homebrew/etc/prometheus.yml

Windows

# Download latest windows-amd64 zip from Prometheus releases
# Extract to C:\prometheus
cd C:\prometheus
.\prometheus.exe --version
.\prometheus.exe --config.file=prometheus.yml
# UI: http://localhost:9090
