Grafana Installation Guide

Ubuntu / Debian

sudo apt-get install -y adduser libfontconfig1 musl
wget https://dl.grafana.com/oss/release/grafana_11.2.0_amd64.deb
sudo dpkg -i grafana_11.2.0_amd64.deb
sudo systemctl enable --now grafana-server
grafana-server -v

Arch Linux

sudo pacman -Syu --noconfirm grafana
sudo systemctl enable --now grafana
grafana-server -v

openSUSE

sudo zypper addrepo https://rpm.grafana.com grafana
sudo rpm --import https://rpm.grafana.com/gpg.key
sudo zypper refresh
sudo zypper install -y grafana
sudo systemctl enable --now grafana-server
grafana-server -v

macOS

brew update
brew install grafana
brew services start grafana
grafana-server -v

Windows

# winget
winget install --id Grafana.GrafanaOSS -e
