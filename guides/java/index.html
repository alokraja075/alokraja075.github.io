Java JDK Installation Guide (JDK 21)

Ubuntu / Debian

sudo apt update
sudo apt install -y openjdk-21-jdk
java -version
javac -version

RHEL / CentOS / Fedora

sudo dnf install -y java-21-openjdk-devel || sudo yum install -y java-21-openjdk-devel
java -version
javac -version

Arch Linux

sudo pacman -Syu --noconfirm jdk21-openjdk
java -version
javac -version

openSUSE

sudo zypper refresh
sudo zypper install -y java-21-openjdk-devel
java -version
javac -version

macOS

brew install openjdk@21
echo 'export PATH="/opt/homebrew/opt/openjdk@21/bin:$PATH"' >> ~/.zshrc
source ~/.zshrc
java -version

Windows

# winget
winget install --id EclipseAdoptium.Temurin.21.JDK -e

# choco
choco install temurin21 -y

java -version
javac -version
