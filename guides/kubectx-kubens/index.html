kubectx & kubens Installation Guide

Ubuntu / Debian

sudo apt update
sudo apt install -y git
sudo git clone https://github.com/ahmetb/kubectx /opt/kubectx
sudo ln -sf /opt/kubectx/kubectx /usr/local/bin/kubectx
sudo ln -sf /opt/kubectx/kubens /usr/local/bin/kubens
kubectx -h
kubens -h

RHEL / CentOS / Fedora

sudo dnf install -y git || sudo yum install -y git
sudo git clone https://github.com/ahmetb/kubectx /opt/kubectx
sudo ln -sf /opt/kubectx/kubectx /usr/local/bin/kubectx
sudo ln -sf /opt/kubectx/kubens /usr/local/bin/kubens
kubectx -h
kubens -h

Arch Linux

sudo pacman -Syu --noconfirm kubectx
kubectx -h
kubens -h

openSUSE

sudo zypper refresh
sudo zypper install -y kubectx
kubectx -h
kubens -h

macOS

brew install kubectx
kubectx -h
kubens -h

Windows

# scoop
scoop install kubectx

# choco (package availability may vary)
choco install kubectx -y

kubectx -h
kubens -h
