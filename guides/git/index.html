Git Installation Guide

Ubuntu / Debian

sudo apt update
sudo apt install -y git
git --version

RHEL / CentOS / Fedora

# Fedora
sudo dnf install -y git

# RHEL/CentOS (if dnf unavailable use yum)
sudo yum install -y git

git --version

Arch Linux

sudo pacman -Syu --noconfirm git
git --version

openSUSE

sudo zypper refresh
sudo zypper install -y git
git --version

macOS

# Option 1: Xcode CLI tools
xcode-select --install

# Option 2: Homebrew
brew install git

git --version

Windows

# winget
winget install --id Git.Git -e

# Chocolatey
choco install git -y

# Scoop
scoop install git

git --version

Post Install

git config --global user.name "Your Name"
git config --global user.email "you@example.com"
git config --global init.defaultBranch main
