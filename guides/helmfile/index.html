Helmfile Installation Guide

Ubuntu / Debian / RHEL / Fedora / openSUSE

VERSION=$(curl -s https://api.github.com/repos/helmfile/helmfile/releases/latest | grep -Po '"tag_name": "\K.*?(?=")')
curl -L "https://github.com/helmfile/helmfile/releases/download/${VERSION}/helmfile_${VERSION#v}_linux_amd64.tar.gz" -o helmfile.tar.gz
tar -xzf helmfile.tar.gz
sudo mv helmfile /usr/local/bin/
helmfile --version

Arch Linux

sudo pacman -Syu --noconfirm helmfile
helmfile --version

macOS

brew install helmfile
helmfile --version

Windows

winget install --id Helmfile.Helmfile -e
# or choco install helmfile -y
helmfile --version
