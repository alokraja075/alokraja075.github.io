AWS CLI v2 Installation Guide

Ubuntu / Debian

sudo apt update
sudo apt install -y unzip curl
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
unzip awscliv2.zip
sudo ./aws/install
aws --version

RHEL / CentOS / Fedora

sudo dnf install -y unzip curl || sudo yum install -y unzip curl
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
unzip awscliv2.zip
sudo ./aws/install
aws --version

Arch Linux

sudo pacman -Syu --noconfirm aws-cli
aws --version

openSUSE

sudo zypper refresh
sudo zypper install -y aws-cli
aws --version

macOS

brew install awscli
aws --version

Windows

# winget
winget install --id Amazon.AWSCLI -e

# choco
choco install awscli -y

aws --version

Configure Profile

aws configure
aws sts get-caller-identity
