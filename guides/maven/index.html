Apache Maven Installation Guide

Ubuntu / Debian

sudo apt update
sudo apt install -y maven
mvn -version

RHEL / CentOS / Fedora

sudo dnf install -y maven || sudo yum install -y maven
mvn -version

Arch Linux

sudo pacman -Syu --noconfirm maven
mvn -version

openSUSE

sudo zypper refresh
sudo zypper install -y maven
mvn -version

macOS

brew install maven
mvn -version

Windows

# winget
winget install --id Apache.Maven -e

# choco
choco install maven -y

mvn -version
